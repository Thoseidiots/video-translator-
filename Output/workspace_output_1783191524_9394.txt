def _escape_ffmpeg_text(text):
    text = text.replace("\\", "\\\\\\\\")
    text = text.replace(":", "\\\\:")
    text = text.replace("{", "\\\\{")
    text = text.replace("}", "\\\\}")
    text = text.replace("'", "")
    return text
